# 生产环境最佳实践

本文档提供在生产环境中部署和运行 cron 调度库的最佳实践，涵盖配置、监控、日志、安全和故障恢复等方面。

## 生产环境配置

### 1. 调度器配置

```go
package main

import (
    "context"
    "log/slog"
    "os"
    "os/signal"
    "syscall"
    "time"

    "github.com/darkit/cron"
    "github.com/darkit/cron/history"
)

func createProductionScheduler() *cron.Cron {
    // 创建结构化日志
    logger := slog.New(slog.NewJSONHandler(os.Stdout, &slog.HandlerOptions{
        Level: slog.LevelInfo,
    }))

    // 自定义日志适配器
    cronLogger := &SlogAdapter{logger: logger}

    // 创建历史记录存储
    storage, err := history.NewFileStorage("/var/lib/cronapp/history")
    if err != nil {
        logger.Error("Failed to create history storage", "error", err)
        os.Exit(1)
    }

    recorder := history.NewHistoryRecorder(storage)

    // 创建上下文，响应系统信号
    ctx, cancel := signal.NotifyContext(
        context.Background(),
        os.Interrupt,
        syscall.SIGTERM,
        syscall.SIGHUP,
    )
    defer cancel()

    // 创建事件钩子
    eventHook := func(ev cron.Event) {
        if ev.End.IsZero() {
            return
        }

        if ev.Success {
            logger.Info("Task completed",
                "task_id", ev.TaskID,
                "duration", ev.Duration,
                "retries", ev.Retries,
            )
        } else {
            logger.Error("Task failed",
                "task_id", ev.TaskID,
                "error", ev.Error,
                "retries", ev.Retries,
            )
        }
    }

    // 创建调度器
    scheduler := cron.New(
        cron.WithLogger(cronLogger),
        cron.WithContext(ctx),
        cron.WithHistoryRecorder(recorder),
        cron.WithEventHook(eventHook),
    )

    return scheduler
}

// SlogAdapter 适配器
type SlogAdapter struct {
    logger *slog.Logger
}

func (l *SlogAdapter) Debugf(format string, args ...any) {
    l.logger.Debug(fmt.Sprintf(format, args...))
}

func (l *SlogAdapter) Infof(format string, args ...any) {
    l.logger.Info(fmt.Sprintf(format, args...))
}

func (l *SlogAdapter) Warnf(format string, args ...any) {
    l.logger.Warn(fmt.Sprintf(format, args...))
}

func (l *SlogAdapter) Errorf(format string, args ...any) {
    l.logger.Error(fmt.Sprintf(format, args...))
}
```

```go
            "priority": "normal",
```

## 监控和告警

### 1. 任务监控

```go
// 定期监控任务状态
func monitorTasks(c *cron.Cron, logger *slog.Logger) {
    ticker := time.NewTicker(time.Minute)
    defer ticker.Stop()

    for range ticker.C {
        stats := c.GetAllStats()

        for taskID, stat := range stats {
            // 检查任务健康状态
            successRate := float64(stat.SuccessCount) / float64(stat.RunCount) * 100

            if successRate < 95 {
                logger.Warn("Task health degraded",
                    "task_id", taskID,
                    "success_rate", successRate,
                    "run_count", stat.RunCount,
                    "fail_count", stat.FailCount,
                )

                // 发送告警
                sendAlert(fmt.Sprintf("Task %s success rate: %.2f%%", taskID, successRate))
            }

            // 检查任务耗时
            if stat.AvgDuration > 5*time.Minute {
                logger.Warn("Task duration too long",
                    "task_id", taskID,
                    "avg_duration", stat.AvgDuration,
                )
            }

            // 检查是否卡住
            if stat.IsRunning && time.Since(stat.LastRun) > 30*time.Minute {
                logger.Error("Task appears to be stuck",
                    "task_id", taskID,
                    "last_run", stat.LastRun,
                )

                // 紧急告警
                sendUrgentAlert(fmt.Sprintf("Task %s is stuck", taskID))
            }
        }
    }
}

func sendAlert(message string) {
    // 发送告警到 Slack、钉钉、邮件等
    log.Printf("ALERT: %s", message)
}

func sendUrgentAlert(message string) {
    // 发送紧急告警，可能需要电话通知
    log.Printf("URGENT ALERT: %s", message)
}
```

```go
import (

    "github.com/prometheus/client_golang/prometheus/promauto"

var (
    taskExecutions = promauto.NewCounterVec(
        prometheus.CounterOpts{
            Name: "cron_task_executions_total",
            Help: "Total number of task executions",
        },
        []string{"task_id", "status"},
    )

    taskDuration = promauto.NewHistogramVec(
        prometheus.HistogramOpts{
            Name:    "cron_task_duration_seconds",
            Help:    "Task execution duration",
            Buckets: prometheus.DefBuckets,
        },
        []string{"task_id"},
    )

    taskRetries = promauto.NewCounterVec(
        prometheus.CounterOpts{
            Name: "cron_task_retries_total",
            Help: "Total number of task retries",
        },
        []string{"task_id"},
    )
)

// 在事件钩子中记录指标
func metricsEventHook(ev cron.Event) {
    if ev.End.IsZero() {
        return
    }

    status := "success"
    if !ev.Success {
        status = "failure"
    }

    taskExecutions.WithLabelValues(ev.TaskID, status).Inc()
    taskDuration.WithLabelValues(ev.TaskID).Observe(ev.Duration.Seconds())

    if ev.Retries > 0 {
        taskRetries.WithLabelValues(ev.TaskID).Add(float64(ev.Retries))
    }
}

// 启动 Prometheus HTTP 服务器
func startMetricsServer(port int) {
    http.Handle("/metrics", promhttp.Handler())
    go func() {
        if err := http.ListenAndServe(fmt.Sprintf(":%d", port), nil); err != nil {
            log.Fatal(err)
        }
    }()
}
```

### 3. 健康检查端点

```go
// 健康检查 HTTP 服务器
func startHealthCheckServer(c *cron.Cron, port int) {
    http.HandleFunc("/health", func(w http.ResponseWriter, r *http.Request) {
        if !c.IsRunning() {
            w.WriteHeader(http.StatusServiceUnavailable)
            json.NewEncoder(w).Encode(map[string]string{
                "status": "unhealthy",
                "reason": "scheduler not running",
            })
            return
        }

        // 检查任务状态
        stats := c.GetAllStats()
        unhealthyTasks := 0

        for _, stat := range stats {
            if stat.RunCount > 0 {
                successRate := float64(stat.SuccessCount) / float64(stat.RunCount) * 100
                if successRate < 80 {
                    unhealthyTasks++
                }
            }
        }

        if unhealthyTasks > 0 {
            w.WriteHeader(http.StatusOK)  // 仍然返回 200，但标记为 degraded
            json.NewEncoder(w).Encode(map[string]interface{}{
                "status":           "degraded",
                "unhealthy_tasks":  unhealthyTasks,
                "total_tasks":      len(stats),
            })
            return
        }

        w.WriteHeader(http.StatusOK)
        json.NewEncoder(w).Encode(map[string]string{
            "status": "healthy",
        })
    })

    http.HandleFunc("/ready", func(w http.ResponseWriter, r *http.Request) {
        if c.IsRunning() {
            w.WriteHeader(http.StatusOK)
            w.Write([]byte("ready"))
        } else {
            w.WriteHeader(http.StatusServiceUnavailable)
            w.Write([]byte("not ready"))
        }
    })

    go func() {
        if err := http.ListenAndServe(fmt.Sprintf(":%d", port), nil); err != nil {
            log.Fatal(err)
        }
    }()
}
```

```go
import "log/slog"

func setupLogging() *slog.Logger {
    var handler slog.Handler

    // 根据环境选择日志格式
    if os.Getenv("ENVIRONMENT") == "production" {
        // 生产环境使用 JSON 格式
        handler = slog.NewJSONHandler(os.Stdout, &slog.HandlerOptions{
            Level: slog.LevelInfo,
            AddSource: true,
        })
    } else {
        // 开发环境使用文本格式
        handler = slog.NewTextHandler(os.Stdout, &slog.HandlerOptions{
            Level: slog.LevelDebug,
        })
    }

    logger := slog.New(handler)
    slog.SetDefault(logger)

    return logger
}
```

```go
func main() {
```

## 完整生产配置示例

```go
package main

import (
    "context"
    "log"
    "log/slog"
    "os"
    "time"

    "github.com/darkit/cron"
    "github.com/darkit/cron/history"
)

func main() {
    // 1. 加载配置
    config, err := loadConfig()
    if err != nil {
        log.Fatal(err)
    }

    // 2. 设置日志
    logger := setupLogging()

    // 3. 创建历史记录存储
    storage, err := history.NewFileStorage(config.HistoryDir)
    if err != nil {
        logger.Error("Failed to create storage", "error", err)
        os.Exit(1)
    }
    defer storage.Close()

    recorder := history.NewHistoryRecorder(storage)
    defer recorder.Close()

    // 4. 创建调度器
    cronLogger := &SlogAdapter{logger: logger}
    eventHook := metricsEventHook

    c := cron.New(
        cron.WithLogger(cronLogger),
        cron.WithHistoryRecorder(recorder),
        cron.WithEventHook(eventHook),
    )

    // 5. 注册任务
    scheduleProductionJobs(c)

    // 6. 定期清理历史记录
    c.Schedule("cleanup-history", "@daily", func(ctx context.Context) {
        before := time.Now().Add(-time.Duration(config.RetentionDays) * 24 * time.Hour)
        deleted, err := c.CleanupHistory(before)
        if err != nil {
            logger.Error("Failed to cleanup history", "error", err)
            return
        }
        logger.Info("Cleaned up history", "deleted", deleted)
    })

    // 7. 启动监控
    go monitorTasks(c, logger)

    // 8. 启动指标服务器
    if config.EnableMetrics {
        startMetricsServer(config.MetricsPort)
    }

    // 9. 启动健康检查服务器
    startHealthCheckServer(c, 8080)

    // 10. 启动调度器
    c.Start()
    defer c.StopGracefully(30 * time.Second)

    logger.Info("Scheduler started successfully")

    // 11. 等待停止信号
    select {}
}
```
