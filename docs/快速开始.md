# 快速开始

5 分钟快速上手 cron 调度库，从安装到运行您的第一个定时任务。

## 安装

使用 Go modules 安装：

```bash
go get github.com/darkit/cron
```

确保您的项目使用 Go 1.23 或更高版本。

## 最简单的例子

创建 `main.go` 文件：

```go
package main

import (
    "context"
    "fmt"
    "time"

    "github.com/darkit/cron"
)

func main() {
    // 1. 创建调度器
    c := cron.New()

    // 2. 添加任务
    c.Schedule("hello", "*/5 * * * * *", func(ctx context.Context) {
        fmt.Println("Hello, Cron! - ", time.Now().Format("15:04:05"))
    })

    // 3. 启动调度器
    c.Start()
    defer c.StopGracefully(5 * time.Second)

    // 4. 保持程序运行
    fmt.Println("调度器已启动，按 Ctrl+C 停止")
    time.Sleep(30 * time.Second)
}
```

运行程序：

```bash
go run main.go
```

输出：

```
调度器已启动，按 Ctrl+C 停止
Hello, Cron! -  10:00:05
Hello, Cron! -  10:00:10
Hello, Cron! -  10:00:15
Hello, Cron! -  10:00:20
Hello, Cron! -  10:00:25
Hello, Cron! -  10:00:30
```

## 常见使用模式

### 1. 使用预定义常量

```go
c := cron.New()

// 每秒执行
c.Schedule("task1", cron.EverySecond, handler)

// 每分钟执行
c.Schedule("task2", cron.EveryMinute, handler)

// 每小时执行
c.Schedule("task3", cron.EveryHour, handler)

// 每天执行
c.Schedule("task4", cron.EveryDay, handler)

// 每 5 秒执行
c.Schedule("task5", cron.Every5Seconds, handler)
```

### 2. 使用描述符

```go
// 每小时执行
c.Schedule("hourly", "@hourly", handler)

// 每天执行
c.Schedule("daily", "@daily", handler)

// 每 10 秒执行
c.Schedule("frequent", "@every 10s", handler)

// 每 5 分钟执行
c.Schedule("regular", "@every 5m", handler)
```

### 3. 使用 cron 表达式

```go
// 6 段表达式（秒级）
c.Schedule("task1", "0 */10 * * * *", handler)  // 每 10 分钟
c.Schedule("task2", "0 0 8 * * *", handler)     // 每天 8 点
c.Schedule("task3", "0 30 9 * * 1-5", handler)  // 工作日 9:30

// 5 段表达式（分钟级）
c.Schedule("task4", "0 8 * * *", handler)       // 每天 8 点
c.Schedule("task5", "0 0 1 * *", handler)       // 每月 1 日
```

### 4. 带配置选项

```go
c := cron.New()

// 超时控制
c.Schedule("timeout-task", "@every 1m", handler, cron.JobOptions{
    Timeout: 30 * time.Second,
})

// 异步执行
c.Schedule("async-task", "@every 1m", handler, cron.JobOptions{
    Async: true,
})

// 限制并发
c.Schedule("limited-task", "@every 10s", handler, cron.JobOptions{
    MaxConcurrent: 3,
})

// 失败重试
c.Schedule("retry-task", "@every 5m", handler, cron.JobOptions{
    MaxRetries:    3,
    RetryInterval: 1 * time.Second,
})

// 指定首次执行时间，并最多执行 3 次
c.Schedule("campaign-task", "@every 10m", handler, cron.JobOptions{
    StartAt: time.Now().Add(30 * time.Minute),
    MaxRuns: 3,
})

// sugar API：指定时间只执行一次
c.ScheduleOnceAt("launch-task", time.Now().Add(10*time.Minute), handler)

// sugar API：从指定时间开始，只执行 3 次
c.ScheduleLimitedFrom("campaign-task-2", "@every 10m", time.Now().Add(30*time.Minute), 3, handler)
```

### 4.1 常见迁移速查

| 需求                    | 原写法                                                    | 推荐写法                   |
| ----------------------- | --------------------------------------------------------- | -------------------------- |
| 指定时间执行一次        | `Schedule(..., JobOptions{StartAt: runAt, MaxRuns: 1})`   | `ScheduleOnceAt(...)`      |
| 从指定时间开始执行 N 次 | `Schedule(..., JobOptions{StartAt: startAt, MaxRuns: n})` | `ScheduleLimitedFrom(...)` |
| 仅执行 N 次             | `Schedule(..., JobOptions{MaxRuns: n})`                   | `ScheduleLimited(...)`     |
| 应用退出释放资源        | `Stop()` + 手动关闭 recorder                              | `Close()`                  |

### 5. 使用 Job 接口

```go
type BackupJob struct {
    path string
}

func (j *BackupJob) Name() string {
    return "backup-job"
}

func (j *BackupJob) Run(ctx context.Context) error {
    fmt.Printf("备份目录: %s\n", j.path)
    // 执行备份逻辑
    return nil
}

func main() {
    c := cron.New()

    job := &BackupJob{path: "/data"}

    // 使用 Job 的 Name() 方法
    c.ScheduleJobByName("@daily", job)

    // 或者显式指定 ID
    c.ScheduleJob("backup", "@daily", job)

    // sugar API：指定时间只执行一次 Job
    c.ScheduleJobOnceAt("backup-once", time.Now().Add(2*time.Hour), job)

    // sugar API：从指定时间开始，只执行 3 次 Job
    c.ScheduleLimitedJobFrom("backup-window", "@every 30m", time.Now().Add(30*time.Minute), 3, job)

    c.Start()
    defer c.Close()

    select {}
}
```

## 实用功能

### 任务管理

```go
c := cron.New()

// 添加任务
c.Schedule("task1", "@every 10s", handler1)
c.Schedule("task2", "@every 20s", handler2)

c.Start()

// 列出所有任务
tasks := c.List()
fmt.Printf("当前任务: %v\n", tasks)

// 查看下次执行时间
nextRun, _ := c.NextRun("task1")
fmt.Printf("task1 下次执行: %v\n", nextRun)

// 获取任务详情
if info, ok := c.GetTask("task1"); ok {
    fmt.Printf("任务状态: 暂停=%v, 运行中=%v\n", info.IsPaused, info.IsRunning)
}

// 获取所有任务详情
allTasks := c.GetAllTasks()
for _, info := range allTasks {
    fmt.Printf("任务 %s: 下次执行 %v\n", info.ID, info.NextRun)
}

// 立即执行任务
c.RunNow("task1")

// 暂停任务
c.Pause("task2")

// 恢复任务
c.Resume("task2")

// 批量操作：暂停/恢复所有任务
c.PauseAll()
c.ResumeAll()

// 关闭并释放托管资源（如 history recorder）
defer c.Close()

// 删除任务
c.Remove("task2")
```

### 查看统计

```go
// 获取任务统计信息
stats, ok := c.GetStats("task1")
if ok {
    fmt.Printf("运行次数: %d\n", stats.RunCount)
    fmt.Printf("成功次数: %d\n", stats.SuccessCount)
    fmt.Printf("失败次数: %d\n", stats.FailCount)
    fmt.Printf("平均耗时: %v\n", stats.AvgDuration)
}

// 获取所有任务统计
allStats := c.GetAllStats()
for id, stats := range allStats {
    fmt.Printf("任务 %s: 运行 %d 次\n", id, stats.RunCount)
}
```

### 优雅停止

```go
import (
    "os"
    "os/signal"
    "syscall"
)

func main() {
    // 响应系统信号
    ctx, cancel := signal.NotifyContext(
        context.Background(),
        os.Interrupt,
        syscall.SIGTERM,
    )
    defer cancel()

    // 绑定到上下文
    c := cron.New(cron.WithContext(ctx))

    c.Schedule("task", "@every 10s", handler)
    c.Start()

    // 等待信号
    <-ctx.Done()
    fmt.Println("收到停止信号，正在优雅关闭...")

    // 等待 5 秒让任务完成
    time.Sleep(5 * time.Second)
}
```

## 完整示例

### 示例 1: 定期备份

```go
package main

import (
    "context"
    "fmt"
    "log"
    "time"

    "github.com/darkit/cron"
)

func backupDatabase(ctx context.Context) {
    fmt.Println("开始备份数据库...")

    // 检查取消信号
    select {
    case <-ctx.Done():
        fmt.Println("备份被取消")
        return
    default:
    }

    // 模拟备份过程
    time.Sleep(2 * time.Second)

    fmt.Println("数据库备份完成")
}

func cleanupOldBackups(ctx context.Context) {
    fmt.Println("清理旧备份文件...")
    time.Sleep(1 * time.Second)
    fmt.Println("清理完成")
}

func main() {
    c := cron.New()

    // 每天 2 点备份数据库
    c.Schedule("backup", "0 0 2 * * *", backupDatabase, cron.JobOptions{
        Timeout: 10 * time.Minute,
        Async:   true,
    })

    // 每周日 3 点清理旧备份
    c.Schedule("cleanup", "0 0 3 * * 0", cleanupOldBackups)

    c.Start()
    defer c.StopGracefully(30 * time.Second)

    log.Println("备份任务已启动")
    select {}
}
```

### 示例 3: 数据同步

```go
package main

import (
    "context"
    "fmt"
    "time"

    "github.com/darkit/cron"
)

type SyncJob struct {
    source string
    target string
}

func NewSyncJob(source, target string) *SyncJob {
    return &SyncJob{
        source: source,
        target: target,
    }
}

func (j *SyncJob) Name() string {
    return fmt.Sprintf("sync-%s-to-%s", j.source, j.target)
}

func (j *SyncJob) Run(ctx context.Context) error {
    fmt.Printf("同步数据: %s -> %s\n", j.source, j.target)

    // 检查上下文
    select {
    case <-ctx.Done():
        return ctx.Err()
    default:
    }

    // 模拟同步过程
    time.Sleep(1 * time.Second)

    fmt.Println("同步完成")
    return nil
}

func main() {
    c := cron.New()

    // 每 5 分钟同步一次
    job := NewSyncJob("database-a", "database-b")
    c.ScheduleJobByName("@every 5m", job, cron.JobOptions{
        Timeout:       2 * time.Minute,
        MaxRetries:    3,
        RetryInterval: 10 * time.Second,
    })

    c.Start()
    defer c.StopGracefully(10 * time.Second)

    fmt.Println("数据同步服务已启动")
    select {}
}
```
